class ListNode [
    fields val:int, next:ListNode
    method getNext() returning ListNode with locals:
        return &this.next
    method getVal() returning int with locals:
        return &this.val
]
class Stack [
    fields list:ListNode
    method push() returning int with locals tmp:ListNode
        tmp = @ListNode
        !tmp.val = 3
        !tmp.next = &this.list
        !this.list = tmp
        return 0
    method pop() returning int with locals tmp:int:
        if (&this.list == 0): {
            return 0
        } else {
            tmp = ^list.getVal()
            return tmp
        }
]

main with stk:Stack:
    stk = @Stack
    !stk.list = 0
    stk.push(1)
    stk.push(3)
    stk.push(5)
    stk.push(9)
    print(^stk.pop())
    print(^stk.pop())
    print(^stk.pop())
    print(^stk.pop())
    print(^stk.pop())
