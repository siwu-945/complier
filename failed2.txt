class ListNode [
    fields val:int, next:ListNode
    method getNext() returning ListNode with locals:
        return &this.next
    method getVal() returning int with locals:
        return &this.val
]
class Stack [
    fields list:ListNode
    method push(v:int) returning int with locals tmp:ListNode
        tmp = @ListNode
        !tmp.val = v
        !tmp.next = &this.list
        !this.list = tmp
        return 0
    method pop() returning int with locals tmp:int:
        if (&this.list == 0): {
            return 0
        } else {
            tmp = ^this.pop()
            return tmp
        }
]

main with stk:Stack,lst:ListNode:
    stk = @Stack
    !stk.list = 0
    stk.push(1)
    stk.push(3)
    stk.push(5)
    stk.push(9)
    print(^stk.pop())
    print(^stk.pop())
    print(^stk.pop())
    print(^stk.pop())
